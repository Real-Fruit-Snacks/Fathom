# Advanced ls Examples

## Sort by size (largest first)
ls -lhS

## Sort by modification time (newest first)
ls -lt

## Show hidden files with details
ls -la

## List only directories
ls -d */

## List files recursively with full paths
ls -R

## List files with color coding
ls --color=auto

## Show file permissions in octal
ls -l | awk '{k=0;for(i=0;i<=8;i++)k+=((substr($1,i+2,1)~/[rwx]/)*(2^(8-i)));if(k)printf("%0o ",k);print}'

## List files modified in the last 24 hours
ls -la --time-style=+%D | grep $(date +%D)

## Show only files (not directories)
ls -p | grep -v /

## List files sorted by extension
ls -X

## Show inode numbers
ls -i

## List files with human-readable sizes and sorted by size
ls -lhS | head -20
